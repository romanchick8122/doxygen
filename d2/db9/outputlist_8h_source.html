512 // For each output format that is enabled (OutputGenerator::isEnabled()) we forward
513 // the method call.
514 // We use C++11 variadic templates and perfect forwarding to implement forall() generically,
515 // and split the types of the methods from the arguments passed to allow implicit conversions.
516 template<typename T,class... Ts,class... As>
517 void forall(void (T::*methodPtr)(Ts...),As&&... args)
518 {
519 for (const auto &og : m_outputs)
520 {
521 if (og->isEnabled()) (og.get()->*methodPtr)(std::forward<As>(args)...);
522 }
523 }
