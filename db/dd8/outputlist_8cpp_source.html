142void OutputList::generateDoc(const QCString &fileName,int startLine,
143 const Definition *ctx,const MemberDef * md,
144 const QCString &docStr,bool indexWords,
145 bool isExample,const QCString &exampleName,
146 bool singleLine,bool linkFromIndex,
147 bool markdownSupport)
148{
149 int count=0;
150 if (docStr.isEmpty()) return;
151
152 for (const auto &og : m_outputs)
153 {
154 if (og->isEnabled()) count++;
155 }
156
157 if (count>0)
158 {
159 // we want to validate irrespective of the number of output formats
160 // specified as:
161 // - when only XML format there should be warnings as well (XML has its own write routines)
162 // - no formats there should be warnings as well
163 auto parser { createDocParser() };
164 auto ast { validatingParseDoc(*parser.get(),
165 fileName,startLine,
166 ctx,md,docStr,indexWords,isExample,exampleName,
167 singleLine,linkFromIndex,markdownSupport) };
168 if (ast) writeDoc(ast.get(),ctx,md,m_id);
169 }
170}

182void OutputList::parseText(const QCString &textStr)
183{
184 int count=0;
185 for (const auto &og : m_outputs)
186 {
187 if (og->isEnabled()) count++;
188 }
189
190 // we want to validate irrespective of the number of output formats
191 // specified as:
192 // - when only XML format there should be warnings as well (XML has its own write routines)
193 // - no formats there should be warnings as well
194 auto parser { createDocParser() };
195 auto textNode { validatingParseText(*parser.get(), textStr) };
196
197 if (textNode && count>0)
198 {
199 for (const auto &og : m_outputs)
200 {
201 if (og->isEnabled()) og->writeDoc(textNode.get(),0,0,m_id);
202 }
203 }
